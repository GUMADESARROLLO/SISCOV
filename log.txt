1 - Crear vista vtWeekPuntos
    --View: vtWeekPuntos
    --DROP VIEW IF EXISTS vtWeekPuntos;

    CREATE DEFINER = 'root'@'localhost' VIEW vtWeekPuntos
    (
        IdTb,
        title,
        puntos,
        `start`
    )
        AS
        select 
        `calendar`.`IdTb` AS `IdTb`,
        `calendar`.`title` AS `title`,
        `calendar`.`puntos` AS `puntos`,
        `calendar`.`start` AS `start` 
        from `calendar`
        WHERE 
        IdTb IN (SELECT T1.IdTb FROM `work` T1 WHERE T1.Activo = 0)
    ;
